% Family relationships

% Facts: parent(Parent, Child)

parent(john, mary).
parent(bob, alice).
parent(bob, tom).
parent(mary, sue).

% Facts: age(Person, Age)
age(bob, 35).
age(mary, 33).
age(alice, 10).
age(tom, 8).
age(sue, 12).

% Rules

% grandparent(X, Z) if X is parent of Y and Y is parent of Z
grandparent(X, Z) :- parent(X, Y), parent(Y, Z).

% sibling(X, Y) if they share a parent (note: includes self!)
sibling(X, Y) :- parent(Z, X), parent(Z, Y).

% ancestor(X, Y) - transitive closure of parent
ancestor(X, Y) :- parent(X, Y).
ancestor(X, Z) :- parent(X, Y), ancestor(Y, Z).
